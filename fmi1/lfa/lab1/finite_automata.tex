\section*{Implementarea unui automat finit}

\begin{verbatim}
Q = {0, 1, 2, 3}
Sigma = {'a', 'b'}
Delta = {
    (0, 'a'): 1,
    (0, 'b'): 0,
    (1, 'a'): 1,
    (1, 'b'): 2,
    (2, 'a'): 1,
    (2, 'b'): 3,
    (3, 'a'): 3,
    (3, 'b'): 3
}
q0 = 0
F = {3}

def accepta_cuvant(w):
q = q0
for lit in w:
    q = Delta.get((q, lit), None)
    if q is None:
        return False
    return q in F

w = 'ababab'
print(accepta_cuvant(w)) # False
w = 'abb'
print(accepta_cuvant(w)) # True
\end{verbatim}
